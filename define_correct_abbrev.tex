\documentclass{scrartcl}
\usepackage[utf8]{inputenc}
\usepackage[english]{babel}

\usepackage{expl3}
\ExplSyntaxOn
\newcommand\latinabbrev[1]{
  % check if next char is a dot
  \peek_meaning:NTF . {%
    #1\@}{%
  % check if else next char is a letter, i.e., has the same catcode as \'a
  \peek_catcode:NTF a {%
    #1.\@}{%
  #1.\@}}}
\ExplSyntaxOff

% the / after abbrev is part of the command,
% it ensures correct spacing after the abbreviation
\def\abbrev/{\latinabbrev{abbrev}}

% Common abbreviations:
\def\cf/{\latinabbrev{cf}}
\def\eg/{\latinabbrev{e.\,g}}
\def\etal/{\latinabbrev{\textit{et~al}}}
\def\etc/{\latinabbrev{etc}}
\def\ie/{\latinabbrev{i.\,e}}

\begin{document}

\begin{itemize}
	\item \cf/, \cf/.
	\item \eg/, \eg/.
	\item \etal/, \etal/.
	\item \etc/, \etc/.
	\item \ie/, \ie/.
\end{itemize}

\end{document}

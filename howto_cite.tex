\documentclass{scrartcl}
\usepackage[utf8]{inputenc}
\usepackage[english]{babel}

\usepackage[%
	% backend=biber should be prefered, but bibtex is still more common
	backend=bibtex8,
	% needed for bibtex, biber can handle utf8 by default
	bibencoding=ascii,
	% set the style according to your needs
	style=numeric,
	% when citing multiple sources, sort these
	sortcites,
	firstinits=true,
	% use et al. only for 11 and more authors in bibliography
	maxbibnames=10,
	% but use et al. for 3 and more authors in text cites
	maxcitenames=2,
]{biblatex}
% https://tex.stackexchange.com/questions/40798/how-do-i-get-et-al-to-appear-in-italics-when-using-textcite-or-citeauthor-w
\DefineBibliographyStrings{english}{%
	andothers = {\em et\addabbrvspace{}al\adddot}
}

% hyperref makes references clickable
\usepackage[%
	% bookmarks for side panel in pdf viewers
	bookmarks,
	% better handling of urls
	breaklinks=true,
	% disable color boxes around clickable links
	colorlinks=true,
	% set clickable links color to something
	linkcolor=black,
	citecolor=black,
	urlcolor=black,
	pdfpagelayout=TwoPageLeft,
]{hyperref}
% nameref is able to cite theorems and stuff by name
\usepackage{nameref}
%load cleveref after hyperref, see sect 12.1 of manual
\usepackage[capitalise]{cleveref}

\addbibresource{howto_cite.bib}
